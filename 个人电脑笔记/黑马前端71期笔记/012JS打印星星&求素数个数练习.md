```js
                //打印以下图形
                      *
                     ***
                    *****
                   *******

        /*
            行数        空格数       星星数
             0           3            1
             1           2            3
             i          3-i          2*i+1
        
        */
        for (var i = 0; i < 4; i++) {
            // 先输出若干个空格
            for (var j = 0; j < 3 - i; j++) {
                document.write("&ensp;");
            }
            // 再输出若干个星星
            for (var k = 0; k < 2 * i + 1; k++) {
                document.write("*");
            }
            // 换行
            document.write("<br />");
        }
   
```

### 三、

```js
//打印以下图形
      *
     * *
    *   *
   *     *
    *   *
     * *
      *  
 // 3 先处理前四行 后处理后三行
        /*
           i=1 空格 3
           i=2   ...2
             3      1
             4      0
        
        */
        //上半部分
        for (var i = 1; i <= 4; i++) {
            // 先打印空格
            for (var j = 1; j <= 4 - i; j++) {
                document.write('&ensp;');
            }
            // 再打印星星
            for (var k = 1; k <= 2 * i - 1; k++) {
                if (k === 1 || k === 2 * i - 1) {
                    document.write("*");
                } else {
                    document.write("&ensp;");
                }
            }
            // 换行
            document.write('<br />');
        }
        //下半部分
        for (var i = 1; i <= 3; i++) {
            // 先打印空格
            for (var j = 1; j <= i; j++) {
                document.write('&ensp;');
            }
            // 再打印星星  1   5    2  3  3 1  7-2*i
            for (var k = 1; k <= 7 - 2 * i; k++) {
                if (k === 1 || k === 7 - 2 * i) {
                    document.write("*");
                } else {
                    document.write("&ensp;");
                }
            }
            // 换行
            document.write('<br />');
        }

```

### 求素数个数

```js
	// 统计100-200之间素数个数 
       var count = 0;// 计数
       for(var i=100; i<=200; i++) {
           var flag = true; // 假定一开始是素数
           for(var j=2; j<i; j++) {  //素数是除了1和它本身能被整除外其他数不能整除
               if(i%j === 0) {  //如果在i到j之间有能够整除的数，则这个数不是素数
                   flag = false;
                   break; //跳出循环
               }
           }
           if(flag) {
               console.log(i);
               count++;
           }
       }
       console.log("100~200之间素数总个数是"+count+"个");
```
