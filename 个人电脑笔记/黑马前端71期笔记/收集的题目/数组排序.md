 let arr = [51, 2, 65, 3, 48, 57];
//冒泡排序
  function bubble(arr) {
    for (let i in arr) {
      for (let j = 0; j < arr.length - i - 1; j++) {
        console.log(j);
        if (arr[j] > arr[j + 1]) {
          [arr[j + 1], arr[j]] = [arr[j], arr[j + 1]]
        }
      }
    }
    return arr
  }
  let res = bubble(arr)
  console.log(res);//[2, 3, 48, 51, 57, 65]
//插入排序
  function insert(arr) {
    let hand = []
    hand.push(arr[0])
    for (let i = 1; i <= arr.length - 1; i++) {
      for (let j = hand.length; j >= 0; j--) {
        if (arr[i] > hand[j]) {
          hand.splice(j + 1, 0, arr[i])
          break
        }
        if (j === 0) {
          hand.unshift(arr[i])
        }
      }
    }
    return hand
  }
  let res1 = insert(arr)
  console.log(res1);//[2, 3, 48, 51, 57, 65]
