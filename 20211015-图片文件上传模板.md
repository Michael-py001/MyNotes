## 图片/视频/文件上传

```js
import Config from '@/utils/Config.js'
// 图片上传
export const ImageUpload = (op = {}) => {
	return new Promise((resolve, reject) => {
		uni.showLoading({
			title: '上传中...',
			mask: true
		})
		let url = `${Config.hostUrl[Config.env]}/Upload/UploadPublicApi/uploadImage`
		let option = Object.assign({
			url,
			file: '',
			name: 'file'
		}, op)
		uni.uploadFile({
			url,
			filePath: option.filePath,
			name: option.name,
			success: (res) => {
				console.log('res--------------', res)
				let data = res.data
				if (typeof data != "object") {
					data = JSON.parse(res.data)
				}

				if (data.data) {
					resolve(data.data)
				} else {
					reject(data)
				}
			},
			fail: err => {
				console.log('Upload fail', err)

				reject(err)
			},
			complete: () => {
				uni.hideLoading()
			}
		})
	})
}
```

### 调用方法

```js
updateCouponPoster() {
    uni.chooseImage({
        count: 1,
        sizeType: ['original', 'compressed'],
        sourceType: ['album'],
        success: (res) => {
            this.$ImageUpload({
                filePath: res.tempFilePaths[0]
            }).then(res => {
                if (res) {
                    this.$Post('Coupon.updateCouponPoster', {
                        id: this.id,
                        poster_url: res.url
                    }).then(res => {
                        const url = this.url
                        this.url = ''
                        setTimeout(() => {
                            this.url = url
                        }, 1000)
                    })
                }
            })
        }
    });
},
```
