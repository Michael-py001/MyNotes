## 邮箱正则

```js
  isEmailAvailable(emailInput) {
         var myreg = /^([a-zA-Z0-9]+[_|\_|\.]?)*[a-zA-Z0-9]+@([a-zA-Z0-9]+[_|\_|\.]?)*[a-zA-Z0-9]+\.[a-zA-Z]{2,3}$/;
         if (!myreg.test(emailInput)) {
             return false;
         }
         else {
             return true;
         }
      },
```

## 手机号

```js
 // 判断是否为手机号
  function isPoneAvailable (pone) {
    var myreg = /^[1][3,4,5,7,8][0-9]{9}$/;
    if (!myreg.test(pone)) {
      return false;
    } else {
      return true;
    }
  }
```
