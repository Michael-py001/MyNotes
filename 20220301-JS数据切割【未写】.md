  swiperList() {
        let num = 6
        let index = 0
        let arr = []
        let length = this.list.length
        for(let i=0;i<length;i++) {
          if(i%num==0 && i!=0) {
             arr.push(this.list.slice(index,i))
             index = i
          }
          if((i+1)===length) {
            arr.push(this.list.slice(index,(i+1)))
          }
        }
        return arr
        console.log("arr:",arr)
      },
